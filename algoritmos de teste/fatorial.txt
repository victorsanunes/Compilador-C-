int calculaFatorial(int n){
    int j;
    int i;
    int fatorial;
    int resultado;
    fatorial = n;
    resultado = 1;
    i = n;
    j = 1;
    while(i >= j){
        resultado = fatorial * resultado;
        i = i - 1;
        fatorial = fatorial - 1;
    }
    return resultado;
}

void main(void){
    int resultado;
    resultado = calculaFatorial(5);
    output(resultado);
}
